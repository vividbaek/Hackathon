방법 1: fs 모듈 래핑 (Node.js)
javascript
// src/integrations/os-guard-sim.js

const fs = require('fs');
const originalReadFile = fs.readFile;
const originalReadFileSync = fs.readFileSync;

const SENSITIVE_FILES = ['.env', '.env.local', 'credentials.json'];

function isSensitive(filepath) {
  return SENSITIVE_FILES.some(f => filepath.includes(f));
}

// readFile 래핑
fs.readFile = function(path, ...args) {
  if (isSensitive(path)) {
    console.log(`🚨 [OS Guard] BLOCKED: ${path}`);
    const callback = args[args.length - 1];
    if (typeof callback === 'function') {
      callback(new Error('Permission denied (OS Guard)'));
    }
    return;
  }
  return originalReadFile.call(this, path, ...args);
};

// readFileSync 래핑
fs.readFileSync = function(path, ...args) {
  if (isSensitive(path)) {
    console.log(`🚨 [OS Guard] BLOCKED: ${path}`);
    throw new Error('Permission denied (OS Guard)');
  }
  return originalReadFileSync.call(this, path, ...args);
};

방법 2: 에이전트 Wrapper (추천)
javascript
// src/integrations/os-guard-sim.js

const { execSync, spawn } = require('child_process');
const { guard } = require('../guard');

const SENSITIVE_PATHS = [
  '.env', '.env.local', '.env.production',
  'credentials.json', 'secrets.json',
  '.ssh/', '.aws/', '.gnupg/'
];

function checkFileAccess(filepath) {
  for (const sensitive of SENSITIVE_PATHS) {
    if (filepath.includes(sensitive)) {
      return {
        allowed: false,
        reason: `Sensitive file: ${sensitive}`
      };
    }
  }
  return { allowed: true };
}

// 명령어 실행 전 파일 접근 체크
function simulateOSGuard(command) {
  // 파일 경로 추출 (간단한 패턴)
  const filePatterns = [
    /cat\s+([^\s|>]+)/,
    /open\(['"]([^'"]+)['"]\)/,
    /readFile\(['"]([^'"]+)['"]\)/,
    /less\s+([^\s]+)/,
    /vim?\s+([^\s]+)/,
  ];
  
  for (const pattern of filePatterns) {
    const match = command.match(pattern);
    if (match) {
      const filepath = match[1];
      const result = checkFileAccess(filepath);
      if (!result.allowed) {
        return {
          blocked: true,
          filepath,
          reason: result.reason
        };
      }
    }
  }
  
  return { blocked: false };
}

module.exports = { simulateOSGuard, checkFileAccess };

시뮬레이션의 한계
우리가 감시할 수 있는 범위만
Node.js fs 래핑 → Node.js만 가능
명령어 파싱 → 텍스트 분석만 가능
못 잡는 케이스
bash
# 에이전트가 이렇게 실행하면?
subprocess.run(['python3', '-c', 'open(".env").read()'])
Shell Guard: "python3 -c ..." 보이지만 내용 분석 어려움
시뮬레이션: python3 프로세스는 우리가 래핑 안 했으니 못 잡음

→ .env 내용 유출됨
EndpointSecurity의 차이
우리가 래핑한 것만 감시 ❌
OS가 모든 파일 접근을 알려줌 ✅
같은 케이스
bash
subprocess.run(['python3', '-c', 'open(".env").read()'])
python3가 .env 열려고 함
     ↓
OS 커널: "잠깐, AUTH_OPEN 이벤트"
     ↓
es-daemon: "이 PID 감시 대상이네, .env는 민감 파일"
     ↓
es-daemon → OS: "DENY"
     ↓
python3: Permission denied

→ .env 접근 차단됨
비교
시뮬레이션	EndpointSecurity
Node.js fs	✅	✅
Shell 명령어	✅	✅
Python open()	❌	✅
Go os.Open()	❌	✅
Rust std::fs	❌	✅
C fopen()	❌	✅
모든 언어	❌	✅
핵심
시뮬레이션: 우리가 감시할 수 있는 범위만
ES: 모든 프로세스의 모든 파일 접근
모든 프로그램은 파일 열 때 OS를 거침
→ OS가 알려주면 다 잡을 수 있음
→ 우회 방법 없음
한 줄 정리
시뮬레이션: "우리가 볼 수 있는 것만"
ES: "OS가 다 알려줌"
발표 포인트
"시뮬레이션은 우리가 래핑한 범위만 감시합니다.
 하지만 실제 OS Integration Layer에서는
 어떤 언어, 어떤 방식으로 파일에 접근해도
 커널 레벨에서 차단됩니다.
 
 이것이 EndpointSecurity가 필요한 이유입니다."
